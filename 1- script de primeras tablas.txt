



-- 1. Marcas de Repuestos
CREATE TABLE Marcas (
    id_marca INT AUTO_INCREMENT PRIMARY KEY,
    nombre_marca VARCHAR(100) NOT NULL,
    activo TINYINT(1) DEFAULT 1
) ENGINE=InnoDB;

-- 2. Marcas de Vehículos
CREATE TABLE Vehiculos_Marcas (
    id_vehiculo_marca INT AUTO_INCREMENT PRIMARY KEY,
    nombre_marca_v VARCHAR(100) NOT NULL,
    activo TINYINT(1) DEFAULT 1
) ENGINE=InnoDB;

-- 3. Modelos de Vehículos
CREATE TABLE Vehiculos_Modelos (
    id_modelo INT AUTO_INCREMENT PRIMARY KEY,
    id_vehiculo_marca INT NOT NULL,
    nombre_modelo VARCHAR(100) NOT NULL,
    activo TINYINT(1) DEFAULT 1,
    CONSTRAINT fk_modelo_marca_v FOREIGN KEY (id_vehiculo_marca) 
        REFERENCES Vehiculos_Marcas(id_vehiculo_marca)
) ENGINE=InnoDB;

-- 4. Motorizaciones
CREATE TABLE Motorizaciones (
    id_motorizacion INT AUTO_INCREMENT PRIMARY KEY,
    nombre_motor VARCHAR(100) NOT NULL,
    descripcion VARCHAR(255),
    activo TINYINT(1) DEFAULT 1
) ENGINE=InnoDB;

-- 5. Categorías
CREATE TABLE Categorias (
    id_categoria INT AUTO_INCREMENT PRIMARY KEY,
    nombre_categoria VARCHAR(100) NOT NULL,
    activo TINYINT(1) DEFAULT 1
) ENGINE=InnoDB;

-- 6. Ubicaciones
CREATE TABLE Ubicaciones (
    id_ubicacion INT AUTO_INCREMENT PRIMARY KEY,
    estanteria VARCHAR(50),
    pasillo VARCHAR(50),
    cajon VARCHAR(50),
    activo TINYINT(1) DEFAULT 1
) ENGINE=InnoDB;

-- 7. Proveedores
CREATE TABLE Proveedores (
    id_proveedor INT AUTO_INCREMENT PRIMARY KEY,
    razon_social VARCHAR(150) NOT NULL,
    cuit VARCHAR(20) UNIQUE,
    activo TINYINT(1) DEFAULT 1
) ENGINE=InnoDB;

-- 8. Repuestos
CREATE TABLE Repuestos (
    id_repuesto INT AUTO_INCREMENT PRIMARY KEY,
    sku VARCHAR(50) UNIQUE NOT NULL,
    cod_oem VARCHAR(100),
    nombre VARCHAR(150) NOT NULL,
    id_marca INT,
    id_categoria INT,
    id_ubicacion INT,
    id_proveedor INT,
    stock_actual INT DEFAULT 0,
    stock_minimo INT DEFAULT 5,
    precio_costo DECIMAL(10, 2),
    fecha_ingreso TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    activo TINYINT(1) DEFAULT 1,
    CONSTRAINT fk_repuesto_marca FOREIGN KEY (id_marca) REFERENCES Marcas(id_marca),
    CONSTRAINT fk_repuesto_categoria FOREIGN KEY (id_categoria) REFERENCES Categorias(id_categoria),
    CONSTRAINT fk_repuesto_ubicacion FOREIGN KEY (id_ubicacion) REFERENCES Ubicaciones(id_ubicacion),
    CONSTRAINT fk_repuesto_proveedor FOREIGN KEY (id_proveedor) REFERENCES Proveedores(id_proveedor)
) ENGINE=InnoDB;

-- 9. Compatibilidad (Tabla final)
CREATE TABLE Compatibilidad_Vehiculos (
    id_compatibilidad INT AUTO_INCREMENT PRIMARY KEY,
    id_repuesto INT NOT NULL,
    id_modelo INT NOT NULL,
    id_motorizacion INT,
    anio_inicio INT,
    anio_fin INT,
    activo TINYINT(1) DEFAULT 1,
    CONSTRAINT fk_comp_repuesto FOREIGN KEY (id_repuesto) REFERENCES Repuestos(id_repuesto) ON DELETE CASCADE,
    CONSTRAINT fk_comp_modelo FOREIGN KEY (id_modelo) REFERENCES Vehiculos_Modelos(id_modelo),
    CONSTRAINT fk_comp_motor FOREIGN KEY (id_motorizacion) REFERENCES Motorizaciones(id_motorizacion)
) ENGINE=InnoDB;
